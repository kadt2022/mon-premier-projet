package com.takibu.identitycore.application.service;

import com.takibu.identitycore.domain.exception.UserCreationException;
import com.takibu.identitycore.domain.model.SpaceId;
import com.takibu.identitycore.domain.model.UserId;
import com.takibu.identitycore.infrastructure.entity.GroupEntity;
import com.takibu.identitycore.infrastructure.entity.GroupMemberEntity;
import com.takibu.identitycore.infrastructure.jpa.repository.GroupJpaRepository;
import com.takibu.identitycore.infrastructure.jpa.repository.GroupMemberJpaRepository;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.time.Clock;
import java.time.Instant;
import java.util.*;
import java.util.stream.Collectors;

@Service
@RequiredArgsConstructor
public class GroupMembershipService {

    private final GroupJpaRepository groupRepo;
    private final GroupMemberJpaRepository groupMemberRepo;
    private final Clock clock;

    @Transactional
    public void addToGroups(SpaceId spaceId, UserId userId, List<String> groupCodes) {
        if (groupCodes == null || groupCodes.isEmpty()) return;

        List<String> cleaned = groupCodes.stream()
                .filter(Objects::nonNull)
                .map(String::trim)
                .filter(s -> !s.isEmpty())
                .distinct()
                .toList();
        if (cleaned.isEmpty()) return;

        UUID space = spaceId.value();
        UUID user  = userId.value();

        // RÃ©sout les groupes par code
        List<GroupEntity> groups = groupRepo.findBySpaceIdAndCodeIn(space, cleaned);
        Map<String, UUID> codeToGroupId = groups.stream()
                .collect(Collectors.toMap(GroupEntity::getCode, GroupEntity::getId));

        // VÃ©rifie les manquants
        List<String> missing = cleaned.stream()
                .filter(code -> !codeToGroupId.containsKey(code))
                .toList();
        if (!missing.isEmpty()) {
            throw new UserCreationException("Unknown group codes in space " + space + ": " + missing);
        }

        // InsÃ¨re les liaisons manquantes
        Instant now = clock.instant();
        List<GroupMemberEntity> toInsert = new ArrayList<>();
        for (String code : cleaned) {
            UUID groupId = codeToGroupId.get(code);
            boolean exists = groupMemberRepo.existsBySpaceIdAndUserIdAndGroupId(space, user, groupId);
            if (!exists) {
                toInsert.add(GroupMemberEntity.builder()
                        // GroupMemberEntity a @UuidGenerator -> on laisse id = null
                        .spaceId(space)
                        .userId(user)
                        .groupId(groupId)
                        .assignedAt(now)
                        .assignedBy(null)
                        .build());
            }
        }
        if (!toInsert.isEmpty()) {
            groupMemberRepo.saveAll(toInsert);
        }
    }
}

package com.takibu.identitycore.application.service;

import com.takibu.identitycore.application.command.CreateUserCommandV2;
import com.takibu.identitycore.application.mapper.UserMapper;
import com.takibu.identitycore.domain.exception.PasswordPolicyViolationException;
import com.takibu.identitycore.domain.exception.SpaceNotFoundException;
import com.takibu.identitycore.domain.exception.UserAlreadyExistsException;
import com.takibu.identitycore.domain.exception.UserCreationException;
import com.takibu.identitycore.domain.model.*;
import com.takibu.identitycore.domain.passwordpolicy.PasswordPolicy;
import com.takibu.identitycore.domain.repository.AccountCredentialsRepository;
import com.takibu.identitycore.domain.repository.AccountRepository;
import com.takibu.identitycore.domain.repository.UserRepository;
import com.takibu.identitycore.interfaces.rest.response.UserResponse;
import com.takibu.identitycore.port.SpaceManagementPort;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import java.time.Clock;
import java.util.Map;
import java.util.Objects;



@Service
@RequiredArgsConstructor
public class UserOnboardingApplicationService {

    private final UserRepository userRepository;
    private final AccountRepository accountRepository;
    private final AccountCredentialsRepository accountCredentialsRepository;
    private final PasswordHasher passwordHasher;
    private final PasswordPolicy passwordPolicy;
    private final SpaceManagementPort spaceManagementPort;
    private final UserMapper userMapper;
    private final RoleAssignmentService roleAssignmentService;
    private final GroupMembershipService groupMembershipService;
    private final Clock clock;


    /**
     * Main V2 flow â€” create or attach Account, then create Space-scoped User.
     */
    @Transactional
    public UserResponse createUserV2(CreateUserCommandV2 c) {
        Objects.requireNonNull(c, "command");
        SpaceId spaceId = new SpaceId(c.spaceId());
        assertSpaceExists(spaceId);

        // Resolve or create Account
        Account account = resolveOrCreateAccount(c);

        // Uniqueness in space
        assertUserNotExistsInSpace(spaceId, account.getId());
        assertUsernameFree(spaceId, c.username());

        // Build & persist User (domain â†’ entity)
        User user = User.create(
                UserId.generate(),
                spaceId,
                account.getId(),
                c.username(),
                c.firstName(),
                c.lastName(),
                UserStatus.PENDING_ACTIVATION,
                false,
                false,
                null,
                clock.instant(),
                clock.instant(),
                c.metadata()
        );
        User saved = userRepository.save(user);

        // Apply initial assignments (roles/groups) â€” placeholders (implement services later)
        roleAssignmentService.assignRoles(spaceId, saved.getId(), c.roleNames());
        groupMembershipService.addToGroups(spaceId, saved.getId(), c.groupCodes());

        // Build response (email from Account)
        return userMapper.toUserResponse(saved, account.getEmail());

    }

    private void assertSpaceExists(SpaceId spaceId) {
        if (!spaceManagementPort.doesSpaceExist(spaceId)) {
            throw new SpaceNotFoundException("Space " + spaceId.value() + " not found");
        }
    }

    private void assertUsernameFree(SpaceId spaceId, String username) {
        if (userRepository.findByUsername(spaceId, username).isPresent()) {
            throw new UserAlreadyExistsException("Username '" + username + "' already exists in this space");
        }
    }


    private void assertUserNotExistsInSpace(SpaceId spaceId, AccountId accountId) {
        if (userRepository.existsBySpaceAndAccount(spaceId, accountId)) {
            throw new UserAlreadyExistsException("Account already has a profile in this space");
        }
    }

    private Account resolveOrCreateAccount(CreateUserCommandV2 c) {
        if (c.accountId() != null) {
            return accountRepository.findById(AccountId.of(c.accountId()))
                    .orElseThrow(() -> new UserCreationException("Account not found: " + c.accountId()));
        }
        // else must provide email + rawPassword
        if (c.email() == null || c.rawPassword() == null) {
            throw new UserCreationException("Either accountId OR (email + rawPassword) must be provided");
        }
        EmailAddress email = new EmailAddress(c.email());
        return accountRepository.findByEmail(email).orElseGet(() -> createAccountWithCredentials(email, c.rawPassword(), Map.of()));
    }

    private Account createAccountWithCredentials(EmailAddress email, String rawPassword, Map<String, Object> metadata) {
        if (!passwordPolicy.isValid(rawPassword)) {
            throw new PasswordPolicyViolationException("Password does not meet policy");
        }
        Account account = Account.create(email, null, null, metadata);
        // Sauvegarde initiale de l'objet Account. C'est la seule fois qu'on le fait.
        account = accountRepository.save(account);

        PasswordHash pwd = PasswordHash.of(passwordHasher.hash(rawPassword), "bcrypt", 1);
        AccountCredentials accountCredentials = AccountCredentials.builder()
                .accountId(account.getId())
                .passwordHash(pwd)
                .passwordUpdatedAt(clock.instant())
                .mustChangeNextLogin(false)
                .failedAttempts(0)
                .lockedUntil(null)
                .createdAt(clock.instant())
                .updatedAt(clock.instant())
                .version(0L)
                .build();
        // Sauvegarde des credentials, sans toucher Ã  l'Account.
        accountCredentialsRepository.save(accountCredentials);
        return account;
    }
}
