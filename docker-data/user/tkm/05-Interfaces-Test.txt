package com.takibu.identitycore.interfaces.rest.controller;

import com.takibu.identitycore.application.command.CreateUserCommandV2;
import com.takibu.identitycore.application.service.UserOnboardingApplicationService;
import com.takibu.identitycore.interfaces.rest.request.CreateUserRequestV2;
import com.takibu.identitycore.interfaces.rest.response.UserResponse;
import jakarta.validation.Valid;
import lombok.RequiredArgsConstructor;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.validation.annotation.Validated;
import org.springframework.web.bind.annotation.*;


import java.util.UUID;


@RestController
@RequestMapping("/api/spaces/{spaceId}/users")
@RequiredArgsConstructor
@Validated
public class UserControllerV2 {

    private final UserOnboardingApplicationService service;

    @PostMapping
    public ResponseEntity<UserResponse> create(@PathVariable("spaceId") UUID spaceId,
                                               @Valid @RequestBody CreateUserRequestV2 req) {
        CreateUserCommandV2 cmd = CreateUserCommandV2.from(spaceId, req);
        UserResponse created = service.createUserV2(cmd);
        return ResponseEntity.status(HttpStatus.CREATED).body(created);
    }
}
